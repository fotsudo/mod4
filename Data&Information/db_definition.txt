CREATE SCHEMA student_conference_db;
CREATE TABLE student_conference_db.teacher(
	id integer PRIMARY KEY,
	name text,
	faculty text,
	research_group text);
CREATE TABLE student_conference_db.student(
	name text,
	id integer PRIMARY KEY,
	program text);
CREATE TABLE student_conference_db.conference_paper(
	id integer PRIMARY KEY,
	topic text,
	supervisor integer,
	supervisee integer,
	title text,
	FOREIGN KEY(supervisor) REFERENCES student_conference_db.teacher(id),
	FOREIGN KEY(supervisee) REFERENCES student_conference_db.student(id)	
);
CREATE TABLE student_conference_db.conference_session(
	id integer PRIMARY KEY,
	supervisee integer,
        title text,
	duration numeric,
	nb_attende integer,
	FOREIGN KEY(supervisee) REFERENCES student_conference_db.student(id)
);
